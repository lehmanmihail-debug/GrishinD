1 создать нвоый массив элементы которого возводяться на два 
map

const array = [-3,-1, 0, 3.14, 10, 20 ]
const newaArray = array.map(element => element * element)
console.log(newaArray)

2 создать новый массив, который содержит все четные числа 
Filter
const array = [-3,-1, 0, 3.14, 10, 20 ]
const newArray = array.filter(element => element % 2 === 0 )
console.log(newArray)

concat слияние массивов 

const arrayFirst = [-3,-1, 0, 3.14, 10, 20 ]
const arraySecond = [ 30, 40, 50]
const arrayThird = [-10, -20, -30]
const mixedArray = arrayFirst.concat(arraySecond).concat(arrayThird)
console.log(mixedArray)
