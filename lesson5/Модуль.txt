! addEventListener ("click",clicker) вешает оброботчик на любую функцию

const clicker = () => {
const first = document.querySelector("div")
first.textContent = "TEST"
first.style.color = "red"
first.style.fontSize = "100px"
}

const mybtn =document.querySelector(".mybtn")
mybtn.addEventListener("click", clicker)

! setAttribute("id", "first") добовляет или класс или айди для элемента в html

const clicker = () => {
const first = document.querySelector("div")
first.textContent = "TEST"
first.style.color = "red"
first.style.fontSize = "100px"
first.setAttribute("id","first")
}

const mybtn =document.querySelector(".mybtn")
mybtn.addEventListener("click", clicker)

! first.removeAttribute("id") удаляет атрибут заданный в скобках
