# конструкция свич и кейс дефолт (switch case default)
явл аналогом цепочки иф элс

с клавиатуры ввод число, если равно 3 то пишет мало, если число равно 4 или 5, программа пишет в точку
если равно 6 то программа пишет много
если число любое другое программа пишет мимо

let a = +prompt()
    switch (a) {
        case 3: alert('мало')
        break
        case 4:
        case 5: alert('в точку')
        break
        case 6: alert('много')
        break
        default: alert('мимо')
    }
